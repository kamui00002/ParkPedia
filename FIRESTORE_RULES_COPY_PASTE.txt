// Firestore Security Rules (Complete Secure Version)
// Copy and paste this into Firebase Console > Firestore Database > Rules
// 重要: このファイルの内容をすべてコピーして、Firebase Consoleに貼り付けてください

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ===========================
    // Common Helper Functions
    // ===========================

    // Check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }

    // Check if authenticated user matches specified userId
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    // Check if authenticated user owns the document
    function isDocumentOwner() {
      return isAuthenticated() && resource.data.userId == request.auth.uid;
    }

    // Check if request userId matches authenticated user
    function hasValidUserId() {
      return request.resource.data.userId == request.auth.uid;
    }

    // Check if userId field is not being changed
    function userIdNotChanged() {
      return request.resource.data.userId == resource.data.userId;
    }

    // Check if createdAt field is not being changed
    function createdAtNotChanged() {
      return request.resource.data.createdAt == resource.data.createdAt;
    }

    // Check if updatedAt is set to server timestamp
    function hasValidUpdatedAt() {
      return request.resource.data.updatedAt == request.time;
    }

    // Check if createdAt is set to server timestamp
    function hasValidCreatedAt() {
      return request.resource.data.createdAt == request.time;
    }

    // Check if user is admin (by checking admins collection)
    // Note: This function can only be used in contexts where the admin document
    // can be accessed (e.g., update/delete operations, not for initial read checks)
    function isAdmin() {
      return isAuthenticated() && 
             exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }

    // ===========================
    // Parks Collection
    // ===========================
    match /parks/{parkId} {
      // Read: Anyone can read
      allow read: if true;

      // Create: Authenticated users only with required fields and validation
      allow create: if isAuthenticated()
        && hasValidUserId()
        && hasValidCreatedAt()
        && !request.resource.data.keys().hasAny(['updatedAt'])
        // Required fields check
        && request.resource.data.keys().hasAll(['name', 'address', 'userId', 'createdAt'])
        // Data type and length validation
        && request.resource.data.name is string
        && request.resource.data.name.size() > 0
        && request.resource.data.name.size() <= 100
        && request.resource.data.address is string
        && request.resource.data.address.size() > 0
        && request.resource.data.address.size() <= 300
        // Latitude validation if present
        && (!request.resource.data.keys().hasAny(['latitude'])
            || (request.resource.data.latitude is number
                && request.resource.data.latitude >= -90
                && request.resource.data.latitude <= 90))
        // Longitude validation if present
        && (!request.resource.data.keys().hasAny(['longitude'])
            || (request.resource.data.longitude is number
                && request.resource.data.longitude >= -180
                && request.resource.data.longitude <= 180))
        // Description validation if present
        && (!request.resource.data.keys().hasAny(['description'])
            || (request.resource.data.description is string
                && request.resource.data.description.size() <= 1000));

      // Update: Owner only, userId and createdAt immutable, updatedAt required
      // Exception: rating and reviewCount can be updated by any authenticated user
      // (for automatic updates when reviews are added/deleted)
      allow update: if isAuthenticated()
        && (
          // Case 1: Only rating and reviewCount are being updated (by any authenticated user)
          // Check that other critical fields haven't changed
          (request.resource.data.rating is number
            && request.resource.data.rating >= 0
            && request.resource.data.rating <= 5
            && request.resource.data.reviewCount is number
            && request.resource.data.reviewCount >= 0
            && request.resource.data.userId == resource.data.userId
            && request.resource.data.name == resource.data.name
            && request.resource.data.address == resource.data.address
            && request.resource.data.createdAt == resource.data.createdAt)
          ||
          // Case 2: Owner updating other fields
          (isDocumentOwner()
            && userIdNotChanged()
            && createdAtNotChanged()
            && hasValidUpdatedAt()
            // Same data validation as create
            && request.resource.data.name is string
            && request.resource.data.name.size() > 0
            && request.resource.data.name.size() <= 100
            && request.resource.data.address is string
            && request.resource.data.address.size() > 0
            && request.resource.data.address.size() <= 300
            && (!request.resource.data.keys().hasAny(['latitude'])
                || (request.resource.data.latitude is number
                    && request.resource.data.latitude >= -90
                    && request.resource.data.latitude <= 90))
            && (!request.resource.data.keys().hasAny(['longitude'])
                || (request.resource.data.longitude is number
                    && request.resource.data.longitude >= -180
                    && request.resource.data.longitude <= 180))
            && (!request.resource.data.keys().hasAny(['description'])
                || (request.resource.data.description is string
                    && request.resource.data.description.size() <= 1000))
          )
        );

      // Delete: Owner only
      allow delete: if isDocumentOwner();

      // Park images subcollection
      match /images/{imageId} {
        // Read: Anyone can read
        allow read: if true;

        // Write: Park owner only
        allow write: if isAuthenticated()
          && get(/databases/$(database)/documents/parks/$(parkId)).data.userId == request.auth.uid;
      }
    }

    // ===========================
    // Reviews Collection
    // ===========================
    match /reviews/{reviewId} {
      // Read: Anyone can read
      allow read: if true;

      // Create: Authenticated users only with required fields and validation
      allow create: if isAuthenticated()
        && hasValidUserId()
        && hasValidCreatedAt()
        && !request.resource.data.keys().hasAny(['updatedAt'])
        // Required fields check
        && request.resource.data.keys().hasAll(['parkId', 'rating', 'userId', 'createdAt'])
        // Verify park exists
        && exists(/databases/$(database)/documents/parks/$(request.resource.data.parkId))
        // Rating validation
        && request.resource.data.rating is number
        && request.resource.data.rating >= 1
        && request.resource.data.rating <= 5
        // Comment validation if present
        && (!request.resource.data.keys().hasAny(['comment'])
            || (request.resource.data.comment is string
                && request.resource.data.comment.size() <= 1000))
        // Title validation if present
        && (!request.resource.data.keys().hasAny(['title'])
            || (request.resource.data.title is string
                && request.resource.data.title.size() <= 100));

      // Update: Owner only, userId/parkId/createdAt immutable, updatedAt required
      allow update: if isDocumentOwner()
        && userIdNotChanged()
        && createdAtNotChanged()
        && hasValidUpdatedAt()
        // parkId is also immutable
        && request.resource.data.parkId == resource.data.parkId
        // Same data validation as create
        && request.resource.data.rating is number
        && request.resource.data.rating >= 1
        && request.resource.data.rating <= 5
        && (!request.resource.data.keys().hasAny(['comment'])
            || (request.resource.data.comment is string
                && request.resource.data.comment.size() <= 1000))
        && (!request.resource.data.keys().hasAny(['title'])
            || (request.resource.data.title is string
                && request.resource.data.title.size() <= 100));

      // Delete: Owner only
      allow delete: if isDocumentOwner();
    }

    // ===========================
    // Favorites Collection
    // ===========================
    match /favorites/{favoriteId} {
      // Get: Authenticated users can only read their own favorites
      allow get: if isAuthenticated()
        && resource.data.userId == request.auth.uid;

      // List: Authenticated users can only list their own favorites
      // Note: This requires the query to include a where clause filtering by userId
      allow list: if isAuthenticated()
        && request.query.limit <= 100;

      // Create: Authenticated users only with required fields and validation
      allow create: if isAuthenticated()
        && hasValidUserId()
        && hasValidCreatedAt()
        && !request.resource.data.keys().hasAny(['updatedAt'])
        // Required fields check
        && request.resource.data.keys().hasAll(['userId', 'parkId', 'type', 'createdAt'])
        // Verify park exists
        && exists(/databases/$(database)/documents/parks/$(request.resource.data.parkId))
        // Type validation
        && request.resource.data.type is string
        && request.resource.data.type in ['favorite', 'visited', 'wantToVisit']
        // visitedAt allowed only for visited type (optional)
        && (!request.resource.data.keys().hasAny(['visitedAt'])
            || (request.resource.data.type == 'visited'
                && request.resource.data.visitedAt == request.time));

      // Update: Owner only, userId/parkId/type immutable, updatedAt required
      allow update: if isDocumentOwner()
        && userIdNotChanged()
        && createdAtNotChanged()
        && hasValidUpdatedAt()
        && request.resource.data.parkId == resource.data.parkId
        && request.resource.data.type == resource.data.type;

      // Delete: Owner only
      allow delete: if isDocumentOwner();
    }

    // ===========================
    // Reports Collection
    // ===========================
    match /reports/{reportId} {
      // Get: Normal users can only read their own reports, admins can read all
      allow get: if isAuthenticated()
        && (isAdmin() || resource.data.reportedBy == request.auth.uid);

      // List: Authenticated users can list their own reports, admins can list all
      // Note: App should filter by reportedBy for non-admin users
      allow list: if isAuthenticated()
        && request.query.limit <= 100;

      // Create: Authenticated users only with required fields and validation
      allow create: if isAuthenticated()
        && hasValidCreatedAt()
        && !request.resource.data.keys().hasAny(['updatedAt'])
        // Required fields check
        && request.resource.data.keys().hasAll(['parkId', 'reviewId', 'reportedBy', 'reason', 'status', 'createdAt'])
        // reportedBy must be current user's UID
        && request.resource.data.reportedBy == request.auth.uid
        // Verify park and review exist
        && exists(/databases/$(database)/documents/parks/$(request.resource.data.parkId))
        && exists(/databases/$(database)/documents/reviews/$(request.resource.data.reviewId))
        // Reason validation
        && request.resource.data.reason is string
        && request.resource.data.reason in ['inappropriate_content', 'spam', 'harassment', 'other']
        // Status must always be 'pending' on creation
        && request.resource.data.status == 'pending'
        // reportedByEmail validation (optional)
        && (!request.resource.data.keys().hasAny(['reportedByEmail'])
            || request.resource.data.reportedByEmail is string)
        // reviewComment validation (optional)
        && (!request.resource.data.keys().hasAny(['reviewComment'])
            || (request.resource.data.reviewComment is string
                && request.resource.data.reviewComment.size() <= 1000));

      // Update: Admin only (for status changes, etc.)
      allow update: if isAdmin()
        && hasValidUpdatedAt()
        && request.resource.data.status in ['pending', 'resolved', 'dismissed']
        && request.resource.data.reportedBy == resource.data.reportedBy
        && request.resource.data.parkId == resource.data.parkId
        && request.resource.data.reviewId == resource.data.reviewId;

      // Delete: Admin only
      allow delete: if isAdmin();
    }

    // ===========================
    // Blocked Users Collection
    // ===========================
    match /blockedUsers/{blockId} {
      // Get: Authenticated users can only read their own blocks
      allow get: if isAuthenticated()
        && resource.data.blockedBy == request.auth.uid;

      // List: Authenticated users can only list their own blocks
      allow list: if isAuthenticated()
        && request.query.limit <= 100;

      // Create: Authenticated users only with required fields and validation
      allow create: if isAuthenticated()
        && hasValidCreatedAt()
        && !request.resource.data.keys().hasAny(['updatedAt'])
        // Required fields check
        && request.resource.data.keys().hasAll(['blockedBy', 'blockedUserId', 'createdAt'])
        // blockedBy must be current user's UID
        && request.resource.data.blockedBy == request.auth.uid
        // Cannot block self
        && request.resource.data.blockedUserId != request.auth.uid
        // blockedUserId validation
        && request.resource.data.blockedUserId is string;

      // Update: Not allowed
      allow update: if false;

      // Delete: Owner only
      allow delete: if isAuthenticated()
        && resource.data.blockedBy == request.auth.uid;
    }

    // ===========================
    // Users Collection
    // ===========================
    match /users/{userId} {
      // Read: Anyone can read public profile
      allow read: if true;

      // Create: Own document only
      allow create: if isOwner(userId)
        && hasValidCreatedAt()
        && !request.resource.data.keys().hasAny(['updatedAt'])
        && request.resource.data.keys().hasAll(['userId', 'createdAt'])
        && request.resource.data.userId == userId;

      // Update: Own document only, userId and createdAt immutable, updatedAt required
      allow update: if isOwner(userId)
        && userIdNotChanged()
        && createdAtNotChanged()
        && hasValidUpdatedAt()
        // Display name validation if present
        && (!request.resource.data.keys().hasAny(['displayName'])
            || (request.resource.data.displayName is string
                && request.resource.data.displayName.size() <= 50))
        // Bio validation if present
        && (!request.resource.data.keys().hasAny(['bio'])
            || (request.resource.data.bio is string
                && request.resource.data.bio.size() <= 500));

      // Delete: Own document only
      allow delete: if isOwner(userId);

      // User private information subcollection
      // Store email, phone, privateData, etc. here
      match /private/{document} {
        // Read and write: Owner only
        allow read, write: if isOwner(userId);
      }

      // User notification settings subcollection
      match /settings/{document} {
        // Read and write: Owner only
        allow read, write: if isOwner(userId);
      }
    }

    // ===========================
    // Admins Collection
    // ===========================
    match /admins/{adminId} {
      // Read: Users can check if they are admin by querying with their UID
      // Get: Users can read their own admin document (if it exists)
      allow get: if isAuthenticated() && request.auth.uid == adminId;
      
      // List: Authenticated users can query to check their own admin status
      // This allows where('userId', '==', currentUser.uid) queries
      allow list: if isAuthenticated()
        && request.query.limit <= 100;

      // Create: Only existing admins can add new admins
      // adminId (document ID) must match userId in the data
      allow create: if isAdmin()
        && hasValidCreatedAt()
        && request.resource.data.userId == adminId
        && request.resource.data.userId is string;

      // Update: Only admins can update
      allow update: if isAdmin();

      // Delete: Only admins can delete
      allow delete: if isAdmin();
    }
  }
}
